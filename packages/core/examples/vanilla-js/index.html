Usage Example

// Import CornerKit
import CornerKit from '@cornerkit/core';

// Create instance
const ck = new CornerKit();

// Apply to single element
ck.apply('#my-element', {
  radius: 20,
  smoothing: 0.8
});

// Apply to multiple elements
ck.applyAll('.buttons', {
  radius: 16,
  smoothing: 0.85
});

// Update existing squircle
ck.update('#my-element', {
  radius: 30
});

// Remove squircle
ck.remove('#my-element');

// Inspect element
const info = ck.inspect('#my-element');
console.log(info); // { config, tier, dimensions }
